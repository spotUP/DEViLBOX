> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port Fons Adriaensen's fil4.lv2 parametric EQ to WASM, replace ParametricEQEffect everywhere, and add a master FX plugin with a visual frequency-response curve.

**Architecture:** Four fil4 DSP headers (iir, filters, hip, lop) wrapped by a thin `fil4_wasm.cpp`, compiled to `public/fil4/Fil4.{js,wasm}`. A new `Fil4EqEffect.ts` (drop-in for `ParametricEQEffect.ts`) drives it from the main thread. The dub bus `returnEQ` and a new `'Fil4EQ'` master FX entry both use this effect. A canvas-based `Fil4EqCurve.tsx` calls WASM `fil4_get_magnitude` for the visual curve.

**Tech Stack:** Emscripten/WASM, C++11, AudioWorklet, Tone.js ToneAudioNode, React canvas, TypeScript.

```cpp
/**
 * fil4_wasm.cpp — WASM wrapper for Fons Adriaensen's fil4 parametric EQ.
 *
 * Exposes 8 filter stages (HP, LoShelf, P0–P3, HiShelf, LP) + magnitude
 * computation for curve display.
 *
 * Processing order: HP → LoShelf → P0 → P1 → P2 → P3 → HiShelf → LP → gain
 *
 * Build: emcmake cmake .. && emmake make  (in fil4-wasm/build/)
 */

#include <cmath>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <emscripten/emscripten.h>

extern "C" {
#include "iir.h"
#include "hip.h"
#include "lop.h"
}
#include "filters.h"

static constexpr int MAX_INSTANCES = 8;
static constexpr int N_PARA = 4;      // parametric band count

// ── Per-channel DSP state ────────────────────────────────────────────────────
struct Fil4Channel {
    IIRProc       ls, hs;      // low/high shelf
    Fil4Paramsect p[N_PARA];   // Regalia-Mitra parametric
    HighPass      hp;
    LowPass       lp;
};

// ── Instance (parameters + per-channel state) ────────────────────────────────
struct Fil4Instance {
    bool  active      = false;
    float sample_rate = 44100.f;

    Fil4Channel ch[2]; // L=0, R=1

    // HP / LP
    struct { bool en; float freq, q; } hp_p = {false, 20.f, 0.7f};
    struct { bool en; float freq, q; } lp_p = {false, 20000.f, 0.7f};

    // Low shelf / High shelf
    struct { bool en; float freq, gain_db, q; } ls_p = {false, 80.f, 0.f, 0.7f};
    struct { bool en; float freq, gain_db, q; } hs_p = {false, 8000.f, 0.f, 0.7f};

    // Parametric bands
    struct { bool en; float freq, bw, gain_db; } p_p[N_PARA] = {
        {false, 200.f,  1.f, 0.f},
        {false, 500.f,  1.f, 0.f},
        {false, 2000.f, 1.f, 0.f},
        {false, 8000.f, 1.f, 0.f},
    };

    float master_gain = 1.f;
};

static Fil4Instance instances[MAX_INSTANCES];

static int find_free() {
    for (int i = 0; i < MAX_INSTANCES; i++)
        if (!instances[i].active) return i;
    return -1;
}

// Convert dB to amplitude factor used by Fil4Paramsect: 10^(dB/40)
// (matches how lv2.c passes gain to proc())
static inline float db_to_para_amp(float db) {
    return powf(10.f, db / 40.f);
}

// Convert bandwidth in octaves to ratio used by Fil4Paramsect: 2^(bw * 0.5)
static inline float bw_to_para_b(float bw) {
    return powf(2.f, bw * 0.5f);
}

// ── Process one channel through all active filters ───────────────────────────
static void process_channel(Fil4Instance& inst, int ch_idx, float* buf, int n) {
    Fil4Channel& ch = inst.ch[ch_idx];
    const float sr = inst.sample_rate;

    if (inst.hp_p.en)
        hip_interpolate(&ch.hp, true, inst.hp_p.freq, inst.hp_p.q);
    hip_compute(&ch.hp, (uint32_t)n, buf);

    if (inst.ls_p.en)
        iir_interpolate(&ch.ls, inst.ls_p.gain_db, inst.ls_p.freq, inst.ls_p.q);
    iir_compute(&ch.ls, (uint32_t)n, buf);

    for (int i = 0; i < N_PARA; i++) {
        if (inst.p_p[i].en) {
            ch.p[i].proc(n, buf,
                inst.p_p[i].freq / sr,
                bw_to_para_b(inst.p_p[i].bw),
                db_to_para_amp(inst.p_p[i].gain_db));
        }
    }

    if (inst.hs_p.en)
        iir_interpolate(&ch.hs, inst.hs_p.gain_db, inst.hs_p.freq, inst.hs_p.q);
    iir_compute(&ch.hs, (uint32_t)n, buf);

    if (inst.lp_p.en)
        lop_interpolate(&ch.lp, true, inst.lp_p.freq, inst.lp_p.q);
    lop_compute(&ch.lp, (uint32_t)n, buf);

    // Master gain
    for (int s = 0; s < n; s++) buf[s] *= inst.master_gain;
}

static void init_channels(Fil4Instance& inst) {
    const float sr = inst.sample_rate;
    for (int c = 0; c < 2; c++) {
        Fil4Channel& ch = inst.ch[c];
        iir_init(&ch.ls, sr);
        iir_init(&ch.hs, sr);
        for (int i = 0; i < N_PARA; i++) ch.p[i].init();
        hip_setup(&ch.hp, sr, inst.hp_p.freq, inst.hp_p.q);
        lop_setup(&ch.lp, sr, inst.lp_p.freq, inst.lp_p.q);
    }
}

// ── Exported API ─────────────────────────────────────────────────────────────

extern "C" {

EMSCRIPTEN_KEEPALIVE int fil4_create(float sample_rate) {
    int h = find_free();
    if (h < 0) return -1;
    Fil4Instance& inst = instances[h];
    inst = Fil4Instance{};
    inst.active = true;
    inst.sample_rate = sample_rate;
    init_channels(inst);
    return h;
}

EMSCRIPTEN_KEEPALIVE void fil4_destroy(int h) {
    if (h >= 0 && h < MAX_INSTANCES) instances[h].active = false;
}

EMSCRIPTEN_KEEPALIVE void fil4_process(int h, float* L, float* R, int n) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    Fil4Instance& inst = instances[h];
    process_channel(inst, 0, L, n);
    process_channel(inst, 1, R, n);
}

EMSCRIPTEN_KEEPALIVE void fil4_set_hp(int h, int en, float freq, float q) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    instances[h].hp_p = { en != 0, freq, q };
}

EMSCRIPTEN_KEEPALIVE void fil4_set_lp(int h, int en, float freq, float q) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    instances[h].lp_p = { en != 0, freq, q };
}

// which: 0 = low shelf, 1 = high shelf
EMSCRIPTEN_KEEPALIVE void fil4_set_shelf(int h, int which, int en,
                                          float freq, float gain_db, float q) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    if (which == 0) instances[h].ls_p = { en != 0, freq, gain_db, q };
    else            instances[h].hs_p = { en != 0, freq, gain_db, q };
}

// band: 0–3
EMSCRIPTEN_KEEPALIVE void fil4_set_band(int h, int band, int en,
                                         float freq, float bw, float gain_db) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    if (band < 0 || band >= N_PARA) return;
    instances[h].p_p[band] = { en != 0, freq, bw, gain_db };
}

EMSCRIPTEN_KEEPALIVE void fil4_set_gain(int h, float gain) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active) return;
    instances[h].master_gain = std::max(0.f, gain);
}

/**
 * Compute magnitude response (dB) at n log-spaced frequencies.
 *
 * Uses impulse response: feeds a unit impulse through a stateless copy of
 * the current filter chain, then evaluates the DFT at each requested frequency.
 * State of the playing instance is never disturbed.
 *
 * log_freqs[]: input — n frequencies in Hz
 * out_db[]:   output — n magnitude values in dB
 */
EMSCRIPTEN_KEEPALIVE void fil4_get_magnitude(int h, float* log_freqs, float* out_db, int n) {
    if (h < 0 || h >= MAX_INSTANCES || !instances[h].active || n <= 0) return;
    Fil4Instance& inst = instances[h];
    const float sr = inst.sample_rate;

    // Impulse response length: 4096 samples covers Q≤8 at 100 Hz
    const int N_IR = 4096;
    float* ir = (float*)calloc(N_IR, sizeof(float));
    if (!ir) { memset(out_db, 0, n * sizeof(float)); return; }
    ir[0] = 1.f;  // unit impulse

    // Temporary channel with fresh (zero) state — same parameters as inst
    Fil4Channel tmp;
    iir_init(&tmp.ls, sr);
    iir_init(&tmp.hs, sr);
    for (int i = 0; i < N_PARA; i++) tmp.p[i].init();
    hip_setup(&tmp.hp, sr, inst.hp_p.freq, inst.hp_p.q);
    lop_setup(&tmp.lp, sr, inst.lp_p.freq, inst.lp_p.q);

    // Run the impulse through all active filters (parameter smoothing off — single-shot)
    if (inst.hp_p.en) {
        hip_interpolate(&tmp.hp, true, inst.hp_p.freq, inst.hp_p.q);
        hip_compute(&tmp.hp, N_IR, ir);
    }
    if (inst.ls_p.en) {
        iir_interpolate(&tmp.ls, inst.ls_p.gain_db, inst.ls_p.freq, inst.ls_p.q);
        iir_compute(&tmp.ls, N_IR, ir);
    }
    for (int i = 0; i < N_PARA; i++) {
        if (inst.p_p[i].en) {
            tmp.p[i].proc(N_IR, ir,
                inst.p_p[i].freq / sr,
                bw_to_para_b(inst.p_p[i].bw),
                db_to_para_amp(inst.p_p[i].gain_db));
        }
    }
    if (inst.hs_p.en) {
        iir_interpolate(&tmp.hs, inst.hs_p.gain_db, inst.hs_p.freq, inst.hs_p.q);
        iir_compute(&tmp.hs, N_IR, ir);
    }
    if (inst.lp_p.en) {
        lop_interpolate(&tmp.lp, true, inst.lp_p.freq, inst.lp_p.q);
        lop_compute(&tmp.lp, N_IR, ir);
    }

    // For each requested frequency, compute DFT at that exact frequency (Goertzel)
    for (int fi = 0; fi < n; fi++) {
        float freq = log_freqs[fi];
        float w = 2.f * (float)M_PI * freq / sr;
        float coeff = 2.f * cosf(w);
        float s1 = 0.f, s2 = 0.f;
        for (int k = 0; k < N_IR; k++) {
            float s0 = ir[k] + coeff * s1 - s2;
            s2 = s1; s1 = s0;
        }
        // |X(f)|² = s1² + s2² - s1*s2*coeff
        float mag2 = s1*s1 + s2*s2 - s1*s2*coeff;
        float mag  = (mag2 > 0.f) ? sqrtf(mag2) : 1e-10f;
        out_db[fi] = 20.f * log10f(mag > 1e-10f ? mag : 1e-10f);
    }

    free(ir);
}

} // extern "C"

/**
 * Fil4.worklet.js — AudioWorklet processor for the fil4 WASM EQ.
 *
 * Message protocol (main thread → worklet):
 *   { type: 'init', wasmBinary, jsCode }
 *   { type: 'set_hp',    enabled, freq, q }
 *   { type: 'set_lp',    enabled, freq, q }
 *   { type: 'set_shelf', which, enabled, freq, gain, q }  // which: 0=lo, 1=hi
 *   { type: 'set_band',  band, enabled, freq, bw, gain }  // band 0–3
 *   { type: 'set_gain',  gain }
 *   { type: 'get_magnitude', id, freqs: Float32Array }    // freqs transferred
 *   { type: 'dispose' }
 *
 * Worklet → main thread:
 *   { type: 'ready' }
 *   { type: 'magnitude_result', id, data: Float32Array }  // data transferred
 */

// Emscripten WASM shims (same pattern as ParametricEQ.worklet.js)
if (typeof URL === 'undefined') {
  globalThis.URL = class URL {
    constructor(path, base) { this.href = base ? (base + '/' + path) : path; }
    toString() { return this.href; }
  };
}
if (typeof globalThis.WorkerGlobalScope === 'undefined') {
  globalThis.WorkerGlobalScope = globalThis.constructor;
}
if (typeof self === 'undefined') {
  globalThis.self = globalThis;
  if (!self.location) self.location = { href: '' };
}

let sharedModule = null;
let sharedModulePromise = null;

async function getOrCreateModule(wasmBinary, jsCode) {
  if (sharedModule) return sharedModule;
  if (sharedModulePromise) return sharedModulePromise;
  sharedModulePromise = (async () => {
    const wrappedCode = `${jsCode}; return createFil4;`;
    const createModule = new Function(wrappedCode)();
    if (!createModule) throw new Error('No createFil4 factory');
    let capturedMemory = null;
    const orig = WebAssembly.instantiate;
    WebAssembly.instantiate = async function (...args) {
      const r = await orig.apply(this, args);
      const inst = r.instance || r;
      if (inst.exports) for (const v of Object.values(inst.exports)) {
        if (v instanceof WebAssembly.Memory) { capturedMemory = v; break; }
      }
      return r;
    };
    let M;
    try { M = await createModule({ wasmBinary }); } finally { WebAssembly.instantiate = orig; }
    if (capturedMemory && !M.wasmMemory) M.wasmMemory = capturedMemory;
    sharedModule = M;
    return M;
  })();
  return sharedModulePromise;
}

let processorRegistered = false;

class Fil4Processor extends AudioWorkletProcessor {
  constructor() {
    super();
    this.handle = -1;
    this.isInitialized = false;
    this.pendingMessages = [];
    this.Module = null;
    this.L_in = 0; this.R_in = 0; this.L_out = 0; this.R_out = 0;
    this.heapL_in = null; this.heapR_in = null;
    this.heapL_out = null; this.heapR_out = null;
    this._wasmMemory = null;
    this.port.onmessage = this.handleMessage.bind(this);
  }

  async handleMessage(event) {
    const { type, ...data } = event.data;
    switch (type) {
      case 'init':   await this.initialize(data); break;
      case 'dispose': this.cleanup(); break;
      case 'get_magnitude':
        if (this.isInitialized) this.computeMagnitude(data.id, data.freqs);
        break;
      default:
        if (this.isInitialized && this.handle >= 0) this.applyCmd(type, data);
        else this.pendingMessages.push(event.data);
        break;
    }
  }

  applyCmd(type, data) {
    const M = this.Module, h = this.handle;
    switch (type) {
      case 'set_hp':    M._fil4_set_hp   (h, data.enabled ? 1 : 0, data.freq, data.q); break;
      case 'set_lp':    M._fil4_set_lp   (h, data.enabled ? 1 : 0, data.freq, data.q); break;
      case 'set_shelf': M._fil4_set_shelf(h, data.which, data.enabled ? 1 : 0, data.freq, data.gain, data.q); break;
      case 'set_band':  M._fil4_set_band (h, data.band,  data.enabled ? 1 : 0, data.freq, data.bw, data.gain); break;
      case 'set_gain':  M._fil4_set_gain (h, data.gain); break;
    }
  }

  computeMagnitude(id, freqs) {
    const M = this.Module, h = this.handle;
    const n = freqs.length;
    // Allocate WASM memory for freq and output arrays
    const freqPtr = M._malloc(n * 4);
    const outPtr  = M._malloc(n * 4);
    try {
      this.refreshHeapViews();
      const heapF32 = new Float32Array(this._wasmMemory ? this._wasmMemory.buffer : M.HEAPF32.buffer);
      heapF32.set(freqs, freqPtr >> 2);
      M._fil4_get_magnitude(h, freqPtr, outPtr, n);
      const result = new Float32Array(heapF32.buffer, outPtr, n).slice();
      this.port.postMessage({ type: 'magnitude_result', id, data: result }, [result.buffer]);
    } finally {
      M._free(freqPtr);
      M._free(outPtr);
    }
  }

  refreshHeapViews() {
    const cur = this._wasmMemory ? this._wasmMemory.buffer
              : (this.Module.HEAPF32 ? this.Module.HEAPF32.buffer : null);
    if (!cur) return;
    if (!this.heapL_in || this.heapL_in.buffer !== cur) {
      this.heapL_in  = new Float32Array(cur, this.L_in,  128);
      this.heapR_in  = new Float32Array(cur, this.R_in,  128);
      this.heapL_out = new Float32Array(cur, this.L_out, 128);
      this.heapR_out = new Float32Array(cur, this.R_out, 128);
    }
  }

  async initialize(data) {
    try {
      const M = await getOrCreateModule(data.wasmBinary, data.jsCode);
      this.handle = M._fil4_create(sampleRate);
      if (this.handle < 0) throw new Error('fil4_create failed');
      const bs = 128 * 4;
      this.L_in  = M._malloc(bs); this.R_in  = M._malloc(bs);
      this.L_out = M._malloc(bs); this.R_out = M._malloc(bs);
      this._wasmMemory = M.wasmMemory || null;
      this.Module = M;
      this.refreshHeapViews();
      this.isInitialized = true;
      for (const msg of this.pendingMessages) this.handleMessage({ data: msg });
      this.pendingMessages = [];
      this.port.postMessage({ type: 'ready' });
    } catch (e) {
      console.error('[Fil4] init error:', e);
      this.port.postMessage({ type: 'error', error: e.message });
    }
  }

  cleanup() {
    if (this.Module && this.handle >= 0) {
      this.Module._fil4_destroy(this.handle);
      this.Module._free(this.L_in); this.Module._free(this.R_in);
      this.Module._free(this.L_out); this.Module._free(this.R_out);
    }
    this.handle = -1; this.isInitialized = false;
  }

  process(inputs, outputs) {
    const input = inputs[0], output = outputs[0];
    const outL = output?.[0], outR = output?.[1] || output?.[0];
    if (!input?.[0]?.length) { outL?.fill(0); outR?.fill(0); return true; }
    const inL = input[0], inR = input[1] || input[0], n = inL.length;

    if (!this.isInitialized || this.handle < 0) {
      outL?.set(inL); outR?.set(inR); return true;
    }

    this.refreshHeapViews();
    this.heapL_in.set(inL.subarray(0, n));
    this.heapR_in.set(inR.subarray(0, n));
    this.Module._fil4_process(this.handle, this.L_in, this.R_in, n);
    // Note: fil4_process modifies L_in/R_in in-place (in-place safe per spec)
    outL.set(this.heapL_in.subarray(0, n));
    outR.set(this.heapR_in.subarray(0, n));
    return true;
  }
}

if (!processorRegistered) {
  registerProcessor('fil4-processor', Fil4Processor);
  processorRegistered = true;
}
```

/**
 * Fil4EqEffect — ToneAudioNode wrapper for the fil4 WASM parametric EQ.
 *
 * Drop-in replacement for ParametricEQEffect. All setB1-B4 aliases are
 * preserved so DubBus.ts requires only a type rename.
 *
 * 8 filter stages: HP → LoShelf → P0–P3 → HiShelf → LP → master gain
 */
import * as Tone from 'tone';
import { getNativeAudioNode } from '@utils/audio-context';

function clamp(v: number, lo: number, hi: number): number {
  return Math.max(lo, Math.min(hi, v));
}

export interface Fil4Params {
  hp:  { enabled: boolean; freq: number; q: number };
  lp:  { enabled: boolean; freq: number; q: number };
  ls:  { enabled: boolean; freq: number; gain: number; q: number };
  hs:  { enabled: boolean; freq: number; gain: number; q: number };
  p:   Array<{ enabled: boolean; freq: number; bw: number; gain: number }>;
  masterGain: number;
}

function defaultParams(): Fil4Params {
  return {
    hp: { enabled: false, freq: 20, q: 0.7 },
    lp: { enabled: false, freq: 20000, q: 0.7 },
    ls: { enabled: false, freq: 80, gain: 0, q: 0.7 },
    hs: { enabled: false, freq: 8000, gain: 0, q: 0.7 },
    p:  [
      { enabled: false, freq: 200,  bw: 1.0, gain: 0 },
      { enabled: false, freq: 500,  bw: 1.0, gain: 0 },
      { enabled: false, freq: 2000, bw: 1.0, gain: 0 },
      { enabled: false, freq: 8000, bw: 1.0, gain: 0 },
    ],
    masterGain: 1.0,
  };
}

type ParamsListener = (params: Fil4Params) => void;

export class Fil4EqEffect extends Tone.ToneAudioNode {
  readonly name = 'Fil4EQ';
  readonly input:  Tone.Gain;
  readonly output: Tone.Gain;

  private passthrough:  Tone.Gain;
  private wetGain:      Tone.Gain;
  private workletNode:  AudioWorkletNode | null = null;
  private isReady = false;
  private pendingCmds: object[] = [];

  private params: Fil4Params = defaultParams();
  private listeners = new Set<ParamsListener>();

  private static wasmBinary:     ArrayBuffer | null = null;
  private static jsCode:         string | null = null;
  private static loadedContexts = new Set<BaseAudioContext>();
  private static initPromises   = new Map<BaseAudioContext, Promise<void>>();

  constructor() {
    super();
    this.input       = new Tone.Gain(1);
    this.output      = new Tone.Gain(1);
    this.passthrough = new Tone.Gain(1);
    this.wetGain     = new Tone.Gain(1);

    // Passthrough (before WASM ready)
    this.input.connect(this.passthrough);
    this.passthrough.connect(this.output);
    this.wetGain.connect(this.output);

    void this._initWorklet();
  }

  // ── Event emitter for curve component ───────────────────────────────────

  on(event: 'params', listener: ParamsListener): this {
    this.listeners.add(listener);
    return this;
  }

  off(event: 'params', listener: ParamsListener): this {
    this.listeners.delete(listener);
    return this;
  }

  private emitParams(): void {
    for (const l of this.listeners) l({ ...this.params });
  }

  // ── Parameter API ────────────────────────────────────────────────────────

  setHP(enabled: boolean, freq: number, q: number): void {
    this.params.hp = { enabled, freq: clamp(freq, 5, 1000), q: clamp(q, 0.1, 4) };
    this._send({ type: 'set_hp', enabled, freq: this.params.hp.freq, q: this.params.hp.q });
    this.emitParams();
  }

  setLP(enabled: boolean, freq: number, q: number): void {
    this.params.lp = { enabled, freq: clamp(freq, 500, 20000), q: clamp(q, 0.1, 4) };
    this._send({ type: 'set_lp', enabled, freq: this.params.lp.freq, q: this.params.lp.q });
    this.emitParams();
  }

  setLowShelf(enabled: boolean, freq: number, gainDb: number, q: number): void {
    this.params.ls = { enabled, freq: clamp(freq, 20, 800), gain: clamp(gainDb, -24, 24), q: clamp(q, 0.1, 2) };
    const p = this.params.ls;
    this._send({ type: 'set_shelf', which: 0, enabled, freq: p.freq, gain: p.gain, q: p.q });
    this.emitParams();
  }

  setHighShelf(enabled: boolean, freq: number, gainDb: number, q: number): void {
    this.params.hs = { enabled, freq: clamp(freq, 1000, 20000), gain: clamp(gainDb, -24, 24), q: clamp(q, 0.1, 2) };
    const p = this.params.hs;
    this._send({ type: 'set_shelf', which: 1, enabled, freq: p.freq, gain: p.gain, q: p.q });
    this.emitParams();
  }

  setBand(band: 0|1|2|3, enabled: boolean, freq: number, bw: number, gainDb: number): void {
    this.params.p[band] = {
      enabled,
      freq:  clamp(freq,   20, 20000),
      bw:    clamp(bw,     0.05, 4.0),
      gain:  clamp(gainDb, -24, 24),
    };
    const p = this.params.p[band];
    this._send({ type: 'set_band', band, enabled, freq: p.freq, bw: p.bw, gain: p.gain });
    this.emitParams();
  }

  setMasterGain(linear: number): void {
    this.params.masterGain = clamp(linear, 0, 4);
    this._send({ type: 'set_gain', gain: this.params.masterGain });
    this.emitParams();
  }

  getParams(): Fil4Params { return { ...this.params }; }

  // ── Curve data ───────────────────────────────────────────────────────────

  getMagnitude(n = 512): Promise<Float32Array> {
    if (!this.workletNode || !this.isReady) {
      // Return flat 0 dB until WASM ready
      return Promise.resolve(new Float32Array(n).fill(0));
    }
    const id = `mag-${Date.now()}-${Math.random().toString(36).slice(2)}`;
    const freqs = new Float32Array(n);
    for (let i = 0; i < n; i++) {
      freqs[i] = 20 * Math.pow(20000 / 20, i / (n - 1));
    }
    return new Promise<Float32Array>((resolve) => {
      const onMsg = (e: MessageEvent) => {
        if (e.data.type === 'magnitude_result' && e.data.id === id) {
          this.workletNode!.port.removeEventListener('message', onMsg);
          resolve(e.data.data as Float32Array);
        }
      };
      this.workletNode!.port.addEventListener('message', onMsg);
      this.workletNode!.port.postMessage({ type: 'get_magnitude', id, freqs }, [freqs.buffer]);
    });
  }

  // ── DubBus backwards-compat aliases (maps to parametric bands 0–3) ──────

  setB1Freq(v: number):  void { const p = this.params.p[0]; this.setBand(0, p.enabled || true, v,       p.bw,   p.gain); }
  setB1Gain(v: number):  void { const p = this.params.p[0]; this.setBand(0, v !== 0,            p.freq, p.bw,   v);      }
  setB1Q(v: number):     void { const p = this.params.p[0]; this.setBand(0, p.enabled,          p.freq, v,      p.gain); }
  setB2Freq(v: number):  void { const p = this.params.p[1]; this.setBand(1, p.enabled || true, v,       p.bw,   p.gain); }
  setB2Gain(v: number):  void { const p = this.params.p[1]; this.setBand(1, v !== 0,            p.freq, p.bw,   v);      }
  setB2Q(v: number):     void { const p = this.params.p[1]; this.setBand(1, p.enabled,          p.freq, v,      p.gain); }
  setB3Freq(v: number):  void { const p = this.params.p[2]; this.setBand(2, p.enabled || true, v,       p.bw,   p.gain); }
  setB3Gain(v: number):  void { const p = this.params.p[2]; this.setBand(2, v !== 0,            p.freq, p.bw,   v);      }
  setB3Q(v: number):     void { const p = this.params.p[2]; this.setBand(2, p.enabled,          p.freq, v,      p.gain); }
  setB4Freq(v: number):  void { const p = this.params.p[3]; this.setBand(3, p.enabled || true, v,       p.bw,   p.gain); }
  setB4Gain(v: number):  void { const p = this.params.p[3]; this.setBand(3, v !== 0,            p.freq, p.bw,   v);      }
  setB4Q(v: number):     void { const p = this.params.p[3]; this.setBand(3, p.enabled,          p.freq, v,      p.gain); }

  // ── Getters (for DubBus reads) ───────────────────────────────────────────
  get b1Freq(): number { return this.params.p[0].freq; }
  get b2Freq(): number { return this.params.p[1].freq; }
  get b2Gain(): number { return this.params.p[1].gain; }
  get b2Q():    number { return this.params.p[1].bw;   }
  get b3Freq(): number { return this.params.p[2].freq; }
  get b4Freq(): number { return this.params.p[3].freq; }
  get b4Gain(): number { return this.params.p[3].gain; }

  // ── Internal ─────────────────────────────────────────────────────────────

  private _send(cmd: object): void {
    if (this.workletNode && this.isReady) {
      this.workletNode.port.postMessage(cmd);
    } else {
      this.pendingCmds.push(cmd);
    }
  }

  private async _initWorklet(): Promise<void> {
    try {
      const rawCtx = Tone.getContext().rawContext as AudioContext;
      await Fil4EqEffect.ensureInitialized(rawCtx);

      this.workletNode = new AudioWorkletNode(rawCtx, 'fil4-processor', {
        numberOfInputs: 1, numberOfOutputs: 1, outputChannelCount: [2],
      });

      this.workletNode.port.onmessage = (ev) => {
        if (ev.data.type === 'ready') {
          this.isReady = true;
          for (const cmd of this.pendingCmds)
            this.workletNode!.port.postMessage(cmd);
          this.pendingCmds = [];
          // Hot-swap: connect WASM, then disconnect passthrough
          try {
            const rawIn  = getNativeAudioNode(this.input)!;
            const rawWet = getNativeAudioNode(this.wetGain)!;
            rawIn.connect(this.workletNode!);
            this.workletNode!.connect(rawWet);
            this.passthrough.gain.value = 0;
            // Keepalive
            const keepalive = rawCtx.createGain();
            keepalive.gain.value = 0;
            this.workletNode!.connect(keepalive);
            keepalive.connect(rawCtx.destination);
          } catch (e) {
            console.warn('[Fil4EQ] hot-swap failed:', e);
          }
        }
      };

      this.workletNode.port.postMessage(
        { type: 'init', wasmBinary: Fil4EqEffect.wasmBinary!, jsCode: Fil4EqEffect.jsCode! },
        [Fil4EqEffect.wasmBinary!.slice(0)],
      );
    } catch (err) {
      console.warn('[Fil4EQ] worklet init failed:', err);
    }
  }

  private static async ensureInitialized(ctx: AudioContext): Promise<void> {
    if (this.loadedContexts.has(ctx)) return;
    const existing = this.initPromises.get(ctx);
    if (existing) return existing;
    const p = (async () => {
      const base = (import.meta as { env?: { BASE_URL?: string } }).env?.BASE_URL ?? '/';
      const [wasmResp, jsResp] = await Promise.all([
        fetch(`${base}fil4/Fil4.wasm`),
        fetch(`${base}fil4/Fil4.js`),
      ]);
      this.wasmBinary = await wasmResp.arrayBuffer();
      let js = await jsResp.text();
      // Strip CommonJS export wrapper that confuses the Function() constructor
      js = js.replace(/if\s*\(typeof exports\s*===\s*"object".*$/s, '');
      this.jsCode = js;
      await ctx.audioWorklet.addModule(`${base}fil4/Fil4.worklet.js`);
      this.loadedContexts.add(ctx);
    })();
    this.initPromises.set(ctx, p);
    return p;
  }

  dispose(): this {
    if (this.workletNode) {
      try { this.workletNode.port.postMessage({ type: 'dispose' }); } catch { /**/ }
      try { this.workletNode.disconnect(); } catch { /**/ }
      this.workletNode = null;
    }
    this.passthrough.dispose();
    this.wetGain.dispose();
    this.input.dispose();
    this.output.dispose();
    this.listeners.clear();
    super.dispose();
    return this;
  }
}

```tsx
/**
 * Fil4EqCurve — Canvas frequency-response display for Fil4EqEffect.
 *
 * Calls effect.getMagnitude(512) via WASM whenever params change.
 * Renders grid, 0-dB line, per-band handles, and combined response curve.
 */
import React, { useRef, useEffect, useCallback, useState } from 'react';
import type { Fil4EqEffect, Fil4Params } from '@/engine/effects/Fil4EqEffect';

const W = 600;
const H = 160;
const PAD_L = 32;
const PAD_R = 8;
const PAD_T = 8;
const PAD_B = 20;
const PLOT_W = W - PAD_L - PAD_R;
const PLOT_H = H - PAD_T - PAD_B;
const DB_MIN = -18;
const DB_MAX = 18;
const FREQ_MIN = 20;
const FREQ_MAX = 20000;
const N_POINTS = 512;

function freqToX(f: number): number {
  return PAD_L + (Math.log10(f / FREQ_MIN) / Math.log10(FREQ_MAX / FREQ_MIN)) * PLOT_W;
}
function dbToY(db: number): number {
  return PAD_T + ((DB_MAX - db) / (DB_MAX - DB_MIN)) * PLOT_H;
}

const GRID_FREQS = [100, 200, 500, 1000, 2000, 5000, 10000];
const GRID_DBS   = [-12, -6, 0, 6, 12];

// Band handle colours (Tailwind CSS variables via getComputedStyle are not
// available in canvas; use hard-coded values that match the theme tokens)
const COLOR_PASS    = '#4F8EF7'; // accent-primary (HP/LP)
const COLOR_SHELF   = '#A855F7'; // accent-secondary (shelves)
const COLOR_PARA    = '#F59E0B'; // accent-warning (parametric)
const COLOR_CURVE   = '#34D399'; // accent-highlight
const COLOR_GRID    = 'rgba(255,255,255,0.08)';
const COLOR_ZERO    = 'rgba(255,255,255,0.20)';
const COLOR_TEXT    = 'rgba(255,255,255,0.35)';
const COLOR_BG      = '#1a1a2e';

interface Props {
  effect: Fil4EqEffect;
  width?: number;
  height?: number;
}

export const Fil4EqCurve: React.FC<Props> = ({ effect, width = W, height = H }) => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [magnitude, setMagnitude] = useState<Float32Array | null>(null);
  const [params, setParams]       = useState<Fil4Params>(effect.getParams());
  const rafRef = useRef<number | null>(null);
  const dirtyRef = useRef(false);

  // Listen for parameter changes
  useEffect(() => {
    const onParams = (p: Fil4Params) => {
      setParams({ ...p });
      dirtyRef.current = true;
    };
    effect.on('params', onParams);
    return () => effect.off('params', onParams);
  }, [effect]);

  // Fetch magnitude when dirty
  useEffect(() => {
    if (!dirtyRef.current) return;
    dirtyRef.current = false;
    effect.getMagnitude(N_POINTS).then(setMagnitude);
  }, [params, effect]);

  // Initial fetch
  useEffect(() => {
    effect.getMagnitude(N_POINTS).then(setMagnitude);
  }, [effect]);

  // Draw
  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    ctx.clearRect(0, 0, width, height);
    ctx.fillStyle = COLOR_BG;
    ctx.fillRect(0, 0, width, height);

    // Grid lines — frequencies
    ctx.strokeStyle = COLOR_GRID;
    ctx.lineWidth = 1;
    ctx.font = `9px monospace`;
    ctx.fillStyle = COLOR_TEXT;
    ctx.textAlign = 'center';
    for (const f of GRID_FREQS) {
      const x = freqToX(f);
      ctx.beginPath(); ctx.moveTo(x, PAD_T); ctx.lineTo(x, PAD_T + PLOT_H); ctx.stroke();
      const label = f >= 1000 ? `${f/1000}k` : `${f}`;
      ctx.fillText(label, x, height - 4);
    }

    // Grid lines — dB
    ctx.textAlign = 'right';
    for (const db of GRID_DBS) {
      const y = dbToY(db);
      ctx.strokeStyle = db === 0 ? COLOR_ZERO : COLOR_GRID;
      ctx.lineWidth = db === 0 ? 1.5 : 1;
      ctx.beginPath(); ctx.moveTo(PAD_L, y); ctx.lineTo(PAD_L + PLOT_W, y); ctx.stroke();
      ctx.fillStyle = COLOR_TEXT;
      ctx.fillText(`${db}`, PAD_L - 4, y + 3);
    }

    // Magnitude curve
    if (magnitude && magnitude.length >= N_POINTS) {
      ctx.strokeStyle = COLOR_CURVE;
      ctx.lineWidth = 2;
      ctx.beginPath();
      for (let i = 0; i < N_POINTS; i++) {
        const f = FREQ_MIN * Math.pow(FREQ_MAX / FREQ_MIN, i / (N_POINTS - 1));
        const x = freqToX(f);
        const db = Math.max(DB_MIN - 2, Math.min(DB_MAX + 2, magnitude[i]));
        const y = dbToY(db);
        if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
      }
      ctx.stroke();
    }

    // Band handles
    const handleBands: Array<{ freq: number; gain: number; color: string; enabled: boolean; label: string }> = [
      { freq: params.hp.freq,  gain: 0,              color: COLOR_PASS,  enabled: params.hp.enabled,  label: 'HP' },
      { freq: params.ls.freq,  gain: params.ls.gain, color: COLOR_SHELF, enabled: params.ls.enabled,  label: 'LS' },
      ...params.p.map((p, i) => ({ freq: p.freq, gain: p.gain, color: COLOR_PARA, enabled: p.enabled, label: `P${i+1}` })),
      { freq: params.hs.freq,  gain: params.hs.gain, color: COLOR_SHELF, enabled: params.hs.enabled,  label: 'HS' },
      { freq: params.lp.freq,  gain: 0,              color: COLOR_PASS,  enabled: params.lp.enabled,  label: 'LP' },
    ];

    for (const band of handleBands) {
      const x = freqToX(band.freq);
      const y = dbToY(band.gain);
      ctx.globalAlpha = band.enabled ? 1 : 0.35;
      ctx.beginPath();
      ctx.arc(x, y, 5, 0, 2 * Math.PI);
      ctx.fillStyle = band.color;
      ctx.fill();
      ctx.strokeStyle = 'rgba(0,0,0,0.5)';
      ctx.lineWidth = 1;
      ctx.stroke();
      ctx.globalAlpha = 1;
    }
  }, [magnitude, params, width, height]);

  useEffect(() => {
    if (rafRef.current !== null) cancelAnimationFrame(rafRef.current);
    rafRef.current = requestAnimationFrame(draw);
    return () => { if (rafRef.current !== null) cancelAnimationFrame(rafRef.current); };
  }, [draw]);

  return (
    <canvas
      ref={canvasRef}
      width={width}
      height={height}
      className="block rounded"
      style={{ background: COLOR_BG }}
    />
  );
};
```

/**
 * Fil4EqPanel — Master FX panel for Fil4EqEffect.
 *
 * Layout:
 *   Top:    Fil4EqCurve (600×160)
 *   Bottom: 8 band columns (HP | LoShelf | P1 P2 P3 P4 | HiShelf | LP)
 *           each with enable toggle, freq knob, gain knob (where applicable),
 *           and Q/BW knob.
 */
import React, { useState, useCallback, useEffect } from 'react';
import { Fil4EqCurve } from './Fil4EqCurve';
import { Knob } from '@components/controls/Knob';
import type { Fil4EqEffect } from '@/engine/effects/Fil4EqEffect';

interface Props {
  effect: Fil4EqEffect;
}

interface BandState {
  enabled: boolean;
  freq: number;
  gain: number;
  q: number;
  bw: number;
}

interface PanelState {
  hp:  BandState;
  lp:  BandState;
  ls:  BandState;
  hs:  BandState;
  p:   BandState[];
  masterGain: number;
}

function paramsToState(p: ReturnType<Fil4EqEffect['getParams']>): PanelState {
  return {
    hp: { enabled: p.hp.enabled, freq: p.hp.freq, gain: 0,       q: p.hp.q, bw: 1 },
    lp: { enabled: p.lp.enabled, freq: p.lp.freq, gain: 0,       q: p.lp.q, bw: 1 },
    ls: { enabled: p.ls.enabled, freq: p.ls.freq, gain: p.ls.gain, q: p.ls.q, bw: 1 },
    hs: { enabled: p.hs.enabled, freq: p.hs.freq, gain: p.hs.gain, q: p.hs.q, bw: 1 },
    p:  p.p.map(b => ({ enabled: b.enabled, freq: b.freq, gain: b.gain, q: 1.0, bw: b.bw })),
    masterGain: p.masterGain,
  };
}

export const Fil4EqPanel: React.FC<Props> = ({ effect }) => {
  const [state, setState] = useState<PanelState>(() => paramsToState(effect.getParams()));

  useEffect(() => {
    const onP = () => setState(paramsToState(effect.getParams()));
    effect.on('params', onP);
    return () => effect.off('params', onP);
  }, [effect]);

  const setHP = useCallback((patch: Partial<BandState>) => {
    setState(s => {
      const next = { ...s.hp, ...patch };
      effect.setHP(next.enabled, next.freq, next.q);
      return { ...s, hp: next };
    });
  }, [effect]);

  const setLP = useCallback((patch: Partial<BandState>) => {
    setState(s => {
      const next = { ...s.lp, ...patch };
      effect.setLP(next.enabled, next.freq, next.q);
      return { ...s, lp: next };
    });
  }, [effect]);

  const setLS = useCallback((patch: Partial<BandState>) => {
    setState(s => {
      const next = { ...s.ls, ...patch };
      effect.setLowShelf(next.enabled, next.freq, next.gain, next.q);
      return { ...s, ls: next };
    });
  }, [effect]);

  const setHS = useCallback((patch: Partial<BandState>) => {
    setState(s => {
      const next = { ...s.hs, ...patch };
      effect.setHighShelf(next.enabled, next.freq, next.gain, next.q);
      return { ...s, hs: next };
    });
  }, [effect]);

  const setParaBand = useCallback((band: 0|1|2|3, patch: Partial<BandState>) => {
    setState(s => {
      const bands = [...s.p];
      bands[band] = { ...bands[band], ...patch };
      const b = bands[band];
      effect.setBand(band, b.enabled, b.freq, b.bw, b.gain);
      return { ...s, p: bands };
    });
  }, [effect]);

  const setGain = useCallback((v: number) => {
    setState(s => { effect.setMasterGain(v); return { ...s, masterGain: v }; });
  }, [effect]);

  const toggleBtn = (on: boolean, onClick: () => void) => (
    <button
      type="button"
      onClick={onClick}
      className={`text-[9px] font-mono px-1 py-0.5 rounded border transition-colors ${
        on
          ? 'border-accent-highlight bg-accent-highlight/20 text-accent-highlight'
          : 'border-dark-borderLight bg-dark-bgTertiary text-text-muted'
      }`}
    >
      {on ? 'ON' : 'OFF'}
    </button>
  );

  const bandCol = (label: string, content: React.ReactNode) => (
    <div className="flex flex-col items-center gap-1 px-2 border-r border-dark-border last:border-0 min-w-[60px]">
      <span className="text-[9px] font-mono text-text-secondary font-bold">{label}</span>
      {content}
    </div>
  );

  return (
    <div className="flex flex-col gap-2 p-2 bg-dark-bgSecondary rounded-lg">
      {/* Curve */}
      <div className="flex items-start gap-2">
        <Fil4EqCurve effect={effect} width={560} height={140} />
        <div className="flex flex-col gap-1 items-center">
          <span className="text-[9px] font-mono text-text-muted">GAIN</span>
          <Knob
            value={state.masterGain}
            min={0} max={2} step={0.01}
            onChange={setGain}
            size={32}
            label="Gain"
          />
          <span className="text-[9px] font-mono text-text-secondary">
            {(20 * Math.log10(Math.max(state.masterGain, 0.001))).toFixed(1)} dB
          </span>
        </div>
      </div>

      {/* Band controls */}
      <div className="flex items-start border border-dark-border rounded overflow-x-auto">
        {/* HP */}
        {bandCol('HP', <>
          <Knob value={state.hp.freq} min={5} max={1000} step={1}
            onChange={v => setHP({ freq: v })} size={28} label="Freq" />
          <Knob value={state.hp.q} min={0.1} max={4} step={0.01}
            onChange={v => setHP({ q: v })} size={28} label="Q" />
          {toggleBtn(state.hp.enabled, () => setHP({ enabled: !state.hp.enabled }))}
        </>)}

        {/* Low Shelf */}
        {bandCol('Lo Shelf', <>
          <Knob value={state.ls.freq} min={20} max={800} step={1}
            onChange={v => setLS({ freq: v })} size={28} label="Freq" />
          <Knob value={state.ls.gain} min={-24} max={24} step={0.5}
            onChange={v => setLS({ gain: v })} size={28} label="Gain" />
          <Knob value={state.ls.q} min={0.1} max={2} step={0.01}
            onChange={v => setLS({ q: v })} size={28} label="Q" />
          {toggleBtn(state.ls.enabled, () => setLS({ enabled: !state.ls.enabled }))}
        </>)}

        {/* Parametric bands */}
        {[0,1,2,3].map(i => (
          bandCol(`P${i+1}`, <>
            <Knob value={state.p[i].freq} min={20} max={20000} step={1}
              onChange={v => setParaBand(i as 0|1|2|3, { freq: v })} size={28} label="Freq" />
            <Knob value={state.p[i].gain} min={-24} max={24} step={0.5}
              onChange={v => setParaBand(i as 0|1|2|3, { gain: v })} size={28} label="Gain" />
            <Knob value={state.p[i].bw} min={0.05} max={4} step={0.05}
              onChange={v => setParaBand(i as 0|1|2|3, { bw: v })} size={28} label="BW" />
            {toggleBtn(state.p[i].enabled, () => setParaBand(i as 0|1|2|3, { enabled: !state.p[i].enabled }))}
          </>)
        ))}

        {/* High Shelf */}
        {bandCol('Hi Shelf', <>
          <Knob value={state.hs.freq} min={1000} max={20000} step={10}
            onChange={v => setHS({ freq: v })} size={28} label="Freq" />
          <Knob value={state.hs.gain} min={-24} max={24} step={0.5}
            onChange={v => setHS({ gain: v })} size={28} label="Gain" />
          <Knob value={state.hs.q} min={0.1} max={2} step={0.01}
            onChange={v => setHS({ q: v })} size={28} label="Q" />
          {toggleBtn(state.hs.enabled, () => setHS({ enabled: !state.hs.enabled }))}
        </>)}

        {/* LP */}
        {bandCol('LP', <>
          <Knob value={state.lp.freq} min={500} max={20000} step={10}
            onChange={v => setLP({ freq: v })} size={28} label="Freq" />
          <Knob value={state.lp.q} min={0.1} max={4} step={0.01}
            onChange={v => setLP({ q: v })} size={28} label="Q" />
          {toggleBtn(state.lp.enabled, () => setLP({ enabled: !state.lp.enabled }))}
        </>)}
      </div>
    </div>
  );
};
