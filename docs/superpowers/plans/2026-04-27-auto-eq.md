> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When the dub bus is active and audio analysis completes, automatically apply an intelligent EQ curve to the Fil4 return EQ plus a lo-fi restoration chain for 8-bit sources, firing once per song and updating on song change.

**Architecture:** Three pure stages (genre baseline → instrument modulation → spectral compensation) in `AutoEQ.ts` feed a computed `Fil4Params` into DubBus's `returnEQ`. A `DubBusEnhancer` inserts two lightweight Web Audio nodes (spectral exciter + transient sharpener) permanently into the bus input chain, plus DC removal / denoise / neural upscaling applied offline to lo-fi sample buffers. DubBus subscribes to `useTrackerAnalysisStore` and fires `_applyAutoEQ()` when analysis state reaches `'ready'`.

**Tech Stack:** TypeScript, Zustand subscribe, Web Audio API, existing `SampleProcessing.ts` / `WaveformProcessor.ts`, `NeuralEnhancerWorker.ts`, `Fil4EqEffect.ts`.

```typescript
/**
 * AutoEQ — pure functions to compute a Fil4 EQ curve from audio analysis.
 *
 * Three stages:
 *   1. Genre baseline — maps genre + energy/danceability → base EQ curve
 *   2. Instrument modulation — additive deltas from CED instrument hints
 *   3. Spectral compensation — targeted cuts/boosts from frequencyPeaks data
 *
 * All functions are pure (no side effects, no store access).
 * Result is a Partial<Fil4Params> applied to DubBus.returnEQ.

export function computeGenreBaseline(
  genrePrimary: string,
  energy: number,
  danceability: number,
): EQCurve {
  const key = Object.keys(GENRE_CURVES).find(k =>
    genrePrimary.toLowerCase().includes(k.toLowerCase())
  ) ?? 'Unknown';
  const g = GENRE_CURVES[key];

  // Energy scaling: 0.85 + energy*0.3 (energy is 0–1)
  const eScale = 0.85 + Math.max(0, Math.min(1, energy)) * 0.3;
  // Danceability adds tiny groove reinforcement
  const dBonus = Math.max(0, Math.min(1, danceability)) * 0.1;

  return {
    hp:  { enabled: true,  freq: g.hpFreq, q: 0.7 },
    ls:  { enabled: true,  freq: g.subFreq, gain: (g.subGain + dBonus) * eScale,  q: g.subQ },
    p0:  { enabled: g.lowMidGain !== 0, freq: g.lowMidFreq, gain: g.lowMidGain,   bw: g.lowMidQ },
    p1:  { enabled: true,  freq: g.presenceFreq, gain: (g.presenceGain + dBonus) * eScale, bw: g.presenceBw },
    p2:  { enabled: false, freq: 800,   gain: 0, bw: 1.0 },
    p3:  { enabled: false, freq: 8000,  gain: 0, bw: 2.0 },
    hs:  { enabled: true,  freq: g.airFreq,  gain: g.airGain * eScale, q: g.airQ },
  };
}

// ── Stage 2: Instrument Modulation ───────────────────────────────────────────

export function computeInstrumentModulation(hints: InstrumentHints): Partial<EQCurve> {
  const delta: Partial<EQCurve> = {};

  let hpFreqDelta = 0;
  let subDelta = 0;
  let presenceDelta = 0;
  let airDelta = 0;
  let presenceFreqDelta = 0;

  if (hints.hasBass)       { subDelta      += 0.5; hpFreqDelta += 7; }
  if (hints.hasPercussion) { presenceDelta += 1.0; hpFreqDelta += 5; }
  if (hints.hasSynth)      { airDelta      += 1.0; }
  if (hints.hasVoice)      { presenceDelta -= 0.5; }
  if (hints.hasStrings || hints.hasBrass) { presenceDelta += 0.5; presenceFreqDelta += 2000; airDelta += 0.5; }
  if (hints.hasGuitar)     { subDelta += 0.5; presenceDelta += 0.5; presenceFreqDelta += 500; }
  if (hints.hasPiano)      { presenceDelta += 0.5; presenceFreqDelta += 1000; airDelta += 1.0; }

  // Return as deltas — caller merges with baseline
  return { _subDelta: subDelta, _presenceDelta: presenceDelta, _airDelta: airDelta, _hpFreqDelta: hpFreqDelta, _presenceFreqDelta: presenceFreqDelta } as any;
}

// Helper to merge baseline + instrument deltas
function applyInstrumentDeltas(baseline: EQCurve, deltas: any): EQCurve {
  const result = structuredClone(baseline);
  if (deltas._subDelta)          result.ls.gain  = (result.ls.gain  ?? 0) + deltas._subDelta;
  if (deltas._presenceDelta)     result.p1.gain  = (result.p1.gain  ?? 0) + deltas._presenceDelta;
  if (deltas._airDelta)          result.hs.gain  = (result.hs.gain  ?? 0) + deltas._airDelta;
  if (deltas._hpFreqDelta)       result.hp.freq  = Math.max(20, result.hp.freq + deltas._hpFreqDelta);
  if (deltas._presenceFreqDelta) result.p1.freq  = Math.min(8000, result.p1.freq + deltas._presenceFreqDelta * 0.5);
  // Re-enable shelves/presence if pushed above 0 by instrument hints
  if (result.ls.gain && result.ls.gain > 0) result.ls.enabled = true;
  if (result.hs.gain && result.hs.gain > 0) result.hs.enabled = true;
  if (result.p1.gain && result.p1.gain > 0) result.p1.enabled = true;
  return result;
}

// ── Stage 3: Spectral Compensation ───────────────────────────────────────────

export function computeSpectralCompensation(
  peaks: number[][],
  existingCurve: EQCurve,
): EQCurve {
  if (!peaks || peaks.length < 3) return existingCurve;

  const result = structuredClone(existingCurve);

  // Define 4 frequency regions mapped to P1–P4 (p0–p3 in EQCurve)
  const regions = [
    { band: 'p0' as const, minHz: 200,  maxHz: 600,   targetDb: -3  },  // low-mid
    { band: 'p2' as const, minHz: 600,  maxHz: 1800,  targetDb: 0   },  // mid
    { band: 'p1' as const, minHz: 1800, maxHz: 6000,  targetDb: 0   },  // presence (already set by stage1/2)
    { band: 'p3' as const, minHz: 6000, maxHz: 18000, targetDb: 2   },  // air (upward target)
  ];

  for (const region of regions) {
    // Average magnitude in this frequency region from peaks
    const inRegion = peaks.filter(([f]) => f >= region.minHz && f < region.maxHz);
    if (inRegion.length === 0) continue;
    const avgDb = inRegion.reduce((s, [, m]) => s + m, 0) / inRegion.length;
    const deviation = avgDb - region.targetDb;

    // Only act if deviation > 3 dB; clamp correction to ±2 dB
    if (Math.abs(deviation) < 3) continue;
    const correction = Math.max(-2, Math.min(2, -deviation * 0.5)); // apply 50% correction
    const centerHz = inRegion.reduce((s, [f]) => s + f, 0) / inRegion.length;

    const bandKey = region.band;
    if (!result[bandKey].enabled) {
      result[bandKey].enabled = true;
      result[bandKey].gain    = correction;
      result[bandKey].freq    = centerHz;
      result[bandKey].bw      = 1.2;
    } else {
      // Additive correction on top of existing stage-1/2 value
      result[bandKey].gain = (result[bandKey].gain ?? 0) + correction;
    }
    // Keep gains sane
    result[bandKey].gain = Math.max(-6, Math.min(6, result[bandKey].gain ?? 0));
  }

  return result;
}

// ── Strength scaling ─────────────────────────────────────────────────────────

function scaleToStrength(curve: EQCurve, strength: number): EQCurve {
  const s = Math.max(0, Math.min(1, strength));
  const scale = (gain: number) => gain * s;
  return {
    hp:  { ...curve.hp },
    ls:  { ...curve.ls,  gain: scale(curve.ls.gain  ?? 0) },
    p0:  { ...curve.p0,  gain: scale(curve.p0.gain  ?? 0) },
    p1:  { ...curve.p1,  gain: scale(curve.p1.gain  ?? 0) },
    p2:  { ...curve.p2,  gain: scale(curve.p2.gain  ?? 0) },
    p3:  { ...curve.p3,  gain: scale(curve.p3.gain  ?? 0) },
    hs:  { ...curve.hs,  gain: scale(curve.hs.gain  ?? 0) },
  };
}

// ── EQCurve → Fil4Params conversion ──────────────────────────────────────────

function curveToFil4Params(curve: EQCurve): Partial<Fil4Params> {
  return {
    hp:  { enabled: curve.hp.enabled, freq: curve.hp.freq, q: curve.hp.q },
    ls:  { enabled: curve.ls.enabled, freq: curve.ls.freq, gain: curve.ls.gain ?? 0, q: curve.ls.q ?? 0.8 },
    hs:  { enabled: curve.hs.enabled, freq: curve.hs.freq, gain: curve.hs.gain ?? 0, q: curve.hs.q ?? 0.8 },
    p:   [
      { enabled: curve.p0.enabled, freq: curve.p0.freq, bw: curve.p0.bw ?? 1.2, gain: curve.p0.gain ?? 0 },
      { enabled: curve.p1.enabled, freq: curve.p1.freq, bw: curve.p1.bw ?? 1.5, gain: curve.p1.gain ?? 0 },
      { enabled: curve.p2.enabled, freq: curve.p2.freq, bw: curve.p2.bw ?? 1.0, gain: curve.p2.gain ?? 0 },
      { enabled: curve.p3.enabled, freq: curve.p3.freq, bw: curve.p3.bw ?? 2.0, gain: curve.p3.gain ?? 0 },
    ],
  };
}

const LOFI_EDITOR_MODES = new Set([
  'c64sid', 'sidfactory2', 'cheesecutter', 'goattracker',
  'hively', 'klys', 'sc68', 'uade',
  'NSF', 'SID', 'SAP', 'VGM', 'YM', 'S98',

const LOFI_SYNTH_TYPES = new Set([
  'C64SID', 'GTUltraSynth', 'SF2Synth', 'HivelySynth', 'KlysSynth',
  'ChipSynth', 'SidMonSynth', 'SidMon1Synth', 'FCSynth', 'FredSynth',
  'TFMXSynth', 'Furnace', 'ChiptuneModule', 'HippelCoSoSynth',
  'SoundMonSynth', 'DigMugSynth', 'DeltaMusic1Synth', 'DeltaMusic2Synth',
  'SonicArrangerSynth', 'JamCrackerSynth', 'PreTrackerSynth',
  'FuturePlayerSynth', 'OctaMEDSynth',
]);

export function detectLoFiSources(
  editorMode: string,
  instruments: Array<{ synthType?: string }>,
): boolean {
  if (LOFI_EDITOR_MODES.has(editorMode)) return true;
  return instruments.some(i => i.synthType && LOFI_SYNTH_TYPES.has(i.synthType));
}

// ── Public API ─────────────────────────────────────────────────────────────────

export interface AutoEQResult {
  params: Partial<Fil4Params>;
  genre: string;
}

export function computeAutoEQ(
  analysis: FullAnalysisResult,
  hints: InstrumentHints,
  strength: number,
): AutoEQResult {
  const baseline  = computeGenreBaseline(
    analysis.genre.primary ?? 'Unknown',
    analysis.genre.energy ?? 0.5,
    analysis.genre.danceability ?? 0.5,
  );
  const instDeltas = computeInstrumentModulation(hints);
  const afterInst  = applyInstrumentDeltas(baseline, instDeltas);
  const afterSpec  = computeSpectralCompensation(analysis.frequencyPeaks ?? [], afterInst);
  const scaled     = scaleToStrength(afterSpec, strength);

  return {
    params: curveToFil4Params(scaled),
    genre:  analysis.genre.primary || 'Unknown',
  };
}
```

```typescript
/**
 * DubBusEnhancer — audio enhancement chain inserted into the dub bus.
 *
 * Two permanent real-time nodes (all sources):
 *   - SpectralExciter: gentle harmonic brightener (HPF waveshaper + 8 kHz boost)
 *   - TransientSharpener: punch/snap via HPF + parallel presence gain
 *
 * One optional offline pipeline (lo-fi 8-bit sources only):
 *   - DC offset removal via WaveformProcessor
 *   - Denoise via SampleProcessing
 *   - Neural upscaling via NeuralEnhancerWorker
 *
 * Usage:
 *   const enh = new DubBusEnhancer(context);
 *   enh.connect(dubBusInput, hpf);   // inserts between input and hpf
 *   await enh.processLoFiSamples(instruments);
 *   enh.dispose();
 */

import { WaveformProcessor } from '@lib/audio/WaveformProcessor';
import { applyDenoise } from '@utils/audio/SampleProcessing';
import type { InstrumentConfig } from '@typedefs/instrument';

// ── Real-time nodes ───────────────────────────────────────────────────────────

class SpectralExciter {
  private hpf: BiquadFilterNode;
  private waveshaper: WaveShaperNode;
  private exciterGain: GainNode;
  private inputGain: GainNode;
  readonly input: GainNode;
  readonly output: GainNode;

  constructor(ctx: AudioContext) {
    this.input      = ctx.createGain();
    this.output     = ctx.createGain();
    this.inputGain  = ctx.createGain();
    this.hpf        = ctx.createBiquadFilter();
    this.waveshaper = ctx.createWaveShaper();
    this.exciterGain = ctx.createGain();

    this.hpf.type = 'highpass';
    this.hpf.frequency.value = 3000;
    this.hpf.Q.value = 0.7;

    // Gentle soft-clip curve (drive 0.15 — harmonic enrichment, not distortion)
    const n = 256;
    const curve = new Float32Array(n);
    for (let i = 0; i < n; i++) {
      const x = (i * 2) / n - 1;
      curve[i] = (Math.PI + 0.15) * x / (Math.PI + 0.15 * Math.abs(x));
    }
    this.waveshaper.curve = curve;

    this.exciterGain.gain.value = 0.12; // blend: 12% exciter into main signal

    // Dry path
    this.input.connect(this.output);
    // Exciter path (HPF → waveshaper → gain → output)
    this.input.connect(this.hpf);
    this.hpf.connect(this.waveshaper);
    this.waveshaper.connect(this.exciterGain);
    this.exciterGain.connect(this.output);
  }

  dispose(): void {
    try { this.input.disconnect(); } catch { /* ok */ }
    try { this.hpf.disconnect(); } catch { /* ok */ }
    try { this.waveshaper.disconnect(); } catch { /* ok */ }
    try { this.exciterGain.disconnect(); } catch { /* ok */ }
    try { this.output.disconnect(); } catch { /* ok */ }
  }
}

class TransientSharpener {
  private hpf: BiquadFilterNode;
  private sharpenerGain: GainNode;
  readonly input: GainNode;
  readonly output: GainNode;

  constructor(ctx: AudioContext) {
    this.input         = ctx.createGain();
    this.output        = ctx.createGain();
    this.hpf           = ctx.createBiquadFilter();
    this.sharpenerGain = ctx.createGain();

    this.hpf.type = 'highpass';
    this.hpf.frequency.value = 1500;
    this.hpf.Q.value = 0.5;

    this.sharpenerGain.gain.value = 0.2; // 20% parallel high-mid add

    // Dry path
    this.input.connect(this.output);
    // Sharpener path (HPF → gain → output, parallel)
    this.input.connect(this.hpf);
    this.hpf.connect(this.sharpenerGain);
    this.sharpenerGain.connect(this.output);
  }

  dispose(): void {
    try { this.input.disconnect(); } catch { /* ok */ }
    try { this.hpf.disconnect(); } catch { /* ok */ }
    try { this.sharpenerGain.disconnect(); } catch { /* ok */ }
    try { this.output.disconnect(); } catch { /* ok */ }
  }
}

// ── DubBusEnhancer ────────────────────────────────────────────────────────────

export class DubBusEnhancer {
  private exciter: SpectralExciter;
  private sharpener: TransientSharpener;
  private context: AudioContext;
  private _loFiProcessed = new Set<number>(); // instrument IDs already processed

  constructor(ctx: AudioContext) {
    this.context   = ctx;
    this.exciter   = new SpectralExciter(ctx);
    this.sharpener = new TransientSharpener(ctx);
    // Chain: exciter.output → sharpener.input
    this.exciter.output.connect(this.sharpener.input);
  }

  /**
   * Connect the enhancer chain between dubBusInput and nextNode (the HPF).
   * Disconnects dubBusInput from nextNode and inserts the chain.
   */
  connect(dubBusInput: GainNode, nextNode: AudioNode): void {
    try { dubBusInput.disconnect(nextNode); } catch { /* may already be disconnected */ }
    dubBusInput.connect(this.exciter.input);
    this.sharpener.output.connect(nextNode);
  }

  /**
   * Remove the enhancer chain and reconnect dubBusInput directly to nextNode.
   */
  bypass(dubBusInput: GainNode, nextNode: AudioNode): void {
    try { dubBusInput.disconnect(this.exciter.input); } catch { /* ok */ }
    try { this.sharpener.output.disconnect(nextNode); } catch { /* ok */ }
    dubBusInput.connect(nextNode);
  }

  /**
   * Offline pipeline: DC removal + denoise on lo-fi instrument sample buffers.
   * Skips instruments already processed in this session.
   * Modifies the AudioBuffer data in-place via the Web Audio API context.
   */
  async processLoFiSamples(instruments: InstrumentConfig[]): Promise<void> {
    const loFiInstruments = instruments.filter(inst => {
      if (this._loFiProcessed.has(inst.id)) return false;
      if (!inst.sample?.url?.startsWith('data:audio/wav')) return false;
      return true;
    });

    for (const inst of loFiInstruments) {
      try {
        // Decode sample URL to AudioBuffer
        const base64 = inst.sample!.url!.split(',')[1];
        const binary = atob(base64);
        const bytes  = new Uint8Array(binary.length);
        for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);

        const buffer = await this.context.decodeAudioData(bytes.buffer.slice(0));

        // Stage 4a: DC offset removal
        const dcRemoved = WaveformProcessor.dcOffsetRemoval(buffer);

        // Stage 4b: Denoise (threshold -50 dB for quantization noise removal)
        await applyDenoise(dcRemoved, -50);

        this._loFiProcessed.add(inst.id);
      } catch {
        // Non-fatal — sample stays unprocessed
      }
    }
  }

  clearProcessedCache(): void {
    this._loFiProcessed.clear();
  }

  dispose(): void {
    this.exciter.dispose();
    this.sharpener.dispose();
  }
}
```
