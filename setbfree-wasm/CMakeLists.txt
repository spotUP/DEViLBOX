cmake_minimum_required(VERSION 3.13)
project(SetBfree C)

set(SETBFREE_ROOT "${CMAKE_SOURCE_DIR}/../third-party/setBfree")

# Core organ engine (memstream.c excluded — only used by lv2.c)
set(CORE_SOURCES
    ${SETBFREE_ROOT}/src/tonegen.c
    ${SETBFREE_ROOT}/src/vibrato.c
    ${SETBFREE_ROOT}/src/midi.c
    ${SETBFREE_ROOT}/src/cfgParser.c
    ${SETBFREE_ROOT}/src/program.c
    ${SETBFREE_ROOT}/src/pgmParser.c
    ${SETBFREE_ROOT}/src/state.c
)

# Leslie speaker
set(WHIRL_SOURCES
    ${SETBFREE_ROOT}/b_whirl/whirl.c
    ${SETBFREE_ROOT}/b_whirl/eqcomp.c
)

# Reverb
set(REVERB_SOURCES
    ${SETBFREE_ROOT}/b_reverb/reverb.c
)

# Overdrive/Preamp (overdrive.c is pre-generated by overmaker)
set(OVERDRIVE_SOURCES
    ${SETBFREE_ROOT}/b_overdrive/overdrive.c
)

# Bridge
set(BRIDGE_SOURCES
    ${CMAKE_SOURCE_DIR}/common/setbfree_bridge.c
)

set(ALL_SOURCES
    ${CORE_SOURCES}
    ${WHIRL_SOURCES}
    ${REVERB_SOURCES}
    ${OVERDRIVE_SOURCES}
    ${BRIDGE_SOURCES}
)

add_executable(SetBfree ${ALL_SOURCES})

target_include_directories(SetBfree PRIVATE
    ${SETBFREE_ROOT}/src
    ${SETBFREE_ROOT}/b_overdrive
    ${SETBFREE_ROOT}/b_reverb
    ${SETBFREE_ROOT}/b_whirl
)

set_target_properties(SetBfree PROPERTIES
    OUTPUT_NAME "SetBfree"
    SUFFIX ".js"
)

target_compile_definitions(SetBfree PRIVATE
    _GNU_SOURCE
    HAVE_MEMSTREAM
)

target_compile_options(SetBfree PRIVATE
    -O2
    -Wno-unused-parameter
    -Wno-sign-compare
    -Wno-implicit-function-declaration
)

target_link_options(SetBfree PRIVATE
    "SHELL:-s MODULARIZE=1"
    "SHELL:-s EXPORT_NAME='createSetBfree'"
    "SHELL:-s EXPORTED_FUNCTIONS=['_setbfree_create','_setbfree_destroy','_setbfree_process','_setbfree_note_on','_setbfree_note_off','_setbfree_set_param','_setbfree_get_param','_setbfree_get_num_params','_setbfree_all_notes_off','_setbfree_set_drawbar','_malloc','_free']"
    "SHELL:-s EXPORTED_RUNTIME_METHODS=['ccall','cwrap']"
    "SHELL:-s ALLOW_MEMORY_GROWTH=1"
    "SHELL:-s INITIAL_MEMORY=16777216"
    "SHELL:-s FILESYSTEM=0"
    "SHELL:-s ENVIRONMENT='web'"
    "SHELL:-s SINGLE_FILE=0"
)
